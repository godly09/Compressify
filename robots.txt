Allow: /

# Directories not to crawl

Disallow: /includes/
Disallow: /tmp/
Disallow: /private/

# Files not to crawl
Disallow: /README.txt
Disallow: /config.ini

# Allow search engines to crawl CSS and JS files for better rendering
Allow: /*.css$
Allow: /*.js$

# Allow image crawling for image search
Allow: /*.jpg$
Allow: /*.png$
Allow: /*.gif$
Allow: /*.webp$
Allow: /*.svg$

# Specific rules for major search engines
User-agent: Googlebot
Allow: /
Crawl-delay: 1

User-agent: Bingbot
Allow: /
Crawl-delay: 1

User-agent: Slurp
Allow: /
Crawl-delay: 1

# Social media crawlers
User-agent: Twitterbot
Allow: /

User-agent: LinkedInBot
Allow: /

User-agent: FacebookExternalHit
Allow: /
Crawl-delay: 2</pre>
